1.2使用构造模式后 #
js
function Person(name, age) {
  this.name = name
  this.age = age
  this.say = function() {
    console.log(this.name + '-' + this.age)
  }
}

const person1 = new Person('Tom', 18)
const person2 = new Person('Jerry', 19)
console.log(person1)
console.log(person2)
person1.say()
person2.say()
